05.2

  • LEAD() | 

    SELECT  userName, addr, height AS "",
     height - (LEAD(height, 1, 0) OVER (ORDER BY height DESC)) AS "   "
 FROM userTbl ;

  • FIRST_VALUE() | 

    SELECT addr, userName, height AS "",
     height - ( FIRST_VALUE(height) OVER (PARTITION BY addr ORDER BY height DESC) ) 
           AS "  "
 FROM userTbl ;

  • CUME_DIST() | 

    SELECT  addr, userName, height AS "",
    (CUME_DIST() OVER (PARTITION BY addr ORDER BY height DESC)) * 100 AS " %"
 FROM userTbl ;
